```

│   └── exception/      # 도메인 예외
├── application/
│   ├── port/
│   │   ├── in/         # UseCase 인터페이스 (Inbound)
│   │   └── out/        # Port 인터페이스 (Outbound)

│   ├── controller/     # REST Controller
│   ├── dto/            # Request/Response DTO
│   └── mapper/         # DTO ↔ Domain 변환
└── infrastructure/

    ├── external/       # 외부 API 클라이언트
    └── config/         # Spring 설정
```

**계층 간 의존 규칙**:
- `domain` → 어떤 계층에도 의존하지 않음 (순수 Kotlin)
- `application` → `domain`만 의존
- `api` → `application` (UseCase 인터페이스를 통해)
- `infrastructure` → `domain`, `application` (Port 구현)

class CareRecordQueryService(
    private val careRecordPort: CareRecordQueryPort
) : CareRecordQueryUseCase {

    override fun findTodayVisits(centerId: Long, caregiverId: Long): List<Visit> =

#### 타입 안전성 규칙

```typescript
// ✅ Good: 명시적 타입, unknown 활용, const assertion
interface CareRecord {
  readonly id: number
  readonly clientName: string
  readonly status: CareRecordStatus
  readonly draft: RecordDraft
  readonly visitDate: string // ISO 8601
}

type CareRecordStatus = 'PROCESSING' | 'DRAFT' | 'SUBMITTED' | 'CONFIRMED' | 'REJECTED'

// API 응답 제네릭
interface ApiResponse<T> {
  readonly success: boolean
  readonly data: T
  readonly error?: ApiError
}

// unknown으로 안전한 에러 핸들링
function handleError(error: unknown): string {
  if (error instanceof ApiError) return error.message
  if (error instanceof Error) return error.message
  return '알 수 없는 오류가 발생했습니다'
}

// ❌ Bad: any 사용, 느슨한 타입
function handleError(error: any): string {  // any 금지
  return error.message  // 런타임 에러 위험
}
```
